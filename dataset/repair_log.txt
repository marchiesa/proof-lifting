function SumProd(amounts: seq<nat>, prices: seq<int>, k: nat): int
  requires |amounts| == |prices|
  requires k <= |amounts|
{
  if k == 0 then 0
  else SumProd(amounts, prices, k - 1) + amounts[k - 1] * prices[k - 1]
}

function TotalShares(amounts: seq<nat>, k: nat): nat
  requires k <= |amounts|
{
  if k == 0 then 0
  else TotalShares(amounts, k - 1) + amounts[k - 1]
}

predicate ValidStrategy(buys: seq<nat>, sells: seq<nat>, r: int, s: seq<int>, b: seq<int>)
{
  |buys| == |s| &&
  |sells| == |b| &&
  SumProd(buys, s, |buys|) <= r &&
  TotalShares(sells, |sells|) <= TotalShares(buys, |buys|)
}

function FinalBourles(buys: seq<nat>, sells: seq<nat>, r: int, s: seq<int>, b: seq<int>): int
  requires |buys| == |s|
  requires |sells| == |b|
{
  r - SumProd(buys, s, |buys|) + SumProd(sells, b, |sells|)
}

predicate IsOptimal(result: int, r: int, s: seq<int>, b: seq<int>)
{
  (exists buyIdx: int, numBuy: int, sellIdx: int, numSell: int |
    0 <= buyIdx < |s| && 0 <= numBuy <= r &&
    0 <= sellIdx < |b| && 0 <= numSell <= r ::
    numBuy * s[buyIdx] <= r &&
    numSell <= numBuy &&
    result == r - numBuy * s[buyIdx] + numSell * b[sellIdx])
  &&
  (forall buyIdx: int, numBuy: int, sellIdx: int, numSell: int |
    0 <= buyIdx < |s| && 0 <= numBuy <= r &&
    0 <= sellIdx < |b| && 0 <= numSell <= r ::
    (numBuy * s[buyIdx] <= r && numSell <= numBuy) ==>
    r - numBuy * s[buyIdx] + numSell * b[sellIdx] <= result)
}

method StockArbitraging(n: int, m: int, r: int, s: seq<int>, b: seq<int>) returns (maxBourles: int)
  requires n == |s| >= 1
  requires m == |b| >= 1
  requires r >= 0
  requires forall i | 0 <= i < |s| :: s[i] > 0
  requires forall j | 0 <= j < |b| :: b[j] > 0
  ensures IsOptimal(maxBourles, r, s, b)
{
  var minS := s[0];
  var i := 1;
  while i < |s|
  {
    if s[i] < minS {
      minS := s[i];
    }
    i := i + 1;
  }

  var maxB := b[0];
  i := 1;
  while i < |b|
  {
    if b[i] > maxB {
      maxB := b[i];
    }
    i := i + 1;
  }

  var profit := r % minS + (r / minS) * maxB;
  if profit > r {
    maxBourles := profit;
  } else {
    maxBourles := r;
  }
}
